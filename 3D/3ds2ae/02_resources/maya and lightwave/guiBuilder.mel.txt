Features:
	There are over 32 different controls to use.
	Including 8 Layouts.
	
	The GUI Builder consists of two parts the controls on the left and the
		code window on the right.
		
	
	Reload : If code is a deleted accidentally by pressing this button code
			window will change back to what it was the last time a window was
			successfully run.
			
	View Code : Pops up a window with all the code including the windows prefs.
	
	Clear : Clears the code window.
	
	Run : Generates a window with what in the code window.

	Absolute :  When unchecked the relative size of the generated window will be
		stored into the width and height preferences.  After the correct size is
		selected then recheck it.  You can only resize through an absolute value
		in the width and height if absolute is checked (default).
	
Usage:

	source guiBuilder.mel
	
	Click on a Add button to add it to the code window.
	
	Make an changes to the code that is necessary.
	
	When ready to see a test GUI just hit RUN and the window will be generated.
	
	Click the view code window copy and paste it where you want it.

global proc mecGuiBuilder()
{
       if(`window -exists mecGuiBWindow`)
       {
                  
 		 
                  deleteUI mecGuiBWindow;
       }
                  
        window -widthHeight 800 600 -sizeable false -mnb 1 -title "The GUI Builder" mecGuiBWindow ;
        
	if(`windowPref -q -exists mecGuiBWindow`)
	{
		windowPref -remove mecGuiBWindow;
        }

        mecGuiBWork();

        showWindow mecGuiBWindow; 
        


}

global proc mecGuiBWork()
{
	rowLayout -nc 2 -w 600 -columnWidth2 325 475 -ad2 1 mecGuiBLayoutRowMain;
        

	//Here is were the tabLayout is going.
	$mecTabL = `tabLayout -w 325 -h 600 mecGuiBtabMain`;

	columnLayout mecGuiBLayoutMain;
	
	//Here is were the window pref will go.
	mecGuiBWindowPref( "mecGuiBLayoutMain" );  //mecGuiBLayoutMain

	setParent $mecTabL;
 	//All the Layouts will be inplanted here
	mecGuiBLayouts( "mecGuiBtabMain" );  //mecGuiBtabMain
	
	setParent $mecTabL;	
	//Here are all the basic Controls.
	mecGuiBControls( "mecGuiBtabMain");
	
	setParent $mecTabL;
	//Fields controls.

	mecGuiBFields("mecGuiBtabMain");

        setParent $mecTabL;
        //Group controls
        mecGuiBGroups("mecGuiBtabMain");
        
        setParent $mecTabL;
        // misc controls
        mecGuiBMisc("mecGuiBtabMain");
	
        setParent mecGuiBLayoutRowMain;
	
	columnLayout;
	scrollField -w 475 -h 500 -wordWrap true -ip 0 mecGuiBSf1;
        text -l " ";
        
        rowLayout -nc 5 -columnWidth5 50 65 50 90 75;
        button -label "Run" -c "mecGuiBBuilder" mecGuiButton2;
        button -label "Reload" -c "mecGuiBReload"
               -ann "This is reload that last successfull run.  This will remove all the present code!!!";
	button -label "Clear" -c "scrollField -e -cl mecGuiBSf1"
	       -ann "This will clear the code window but won't reset the reload button.";

        button -label "View Code" -c "mecGuiBViewCode"
		-ann "This will pop up a window with the complete code for the window.";

        setParent..;

	tabLayout -e 
		-tabLabel mecGuiBLayoutMain "Window Prefs" //mecGuiBFieldsSLMain
		-tabLabel mecGuiBcolLayout2 "Layouts"
		-tabLabel mecGuiBControlsColMain2 "Controls"
		-tabLabel mecGuiBFieldsColMain2 "Fields"
		-tabLabel mecGuiBGroupsColMain2 "Groups"
		-tabLabel mecGuiBMiscColMain2 "Misc"
		$mecTabL;    

        text -l "" -vis false mecGuiBReload;
        text -l "None" -vis false mecGuiBError;
}

global proc mecGuiBViewCode()
{
	if(`window -exists mecWinBViewCodeWin`)
		deleteUI mecWinBViewCodeWin;
	if( `windowPref -exists mecWinBViewCodeWin` )
		windowPref -r mecWinBViewCodeWin;

	window -w 620 -h 550 -t "GUI Code View" -sizeable 0 mecWinBViewCodeWin;
	
	//Gathering window information.
	

	string $mecTitle = `textFieldGrp -q -text mecGuiBtfgTitle`;
	int $mecWidth = `intFieldGrp -q -v1 mecGuiBifgWidth`;
	int $mecHeight = `intFieldGrp -q -v1 mecGuiBifgHeight`;
	int $mecSizeable = `checkBox -q -v mecGuiBifgSizeable`;
	string $mecObjName = `textFieldGrp -q -text mecGuiBWinObj`;
	
	int $mecTitleB = `checkBox -q -v mecGuiBifgTitleB`;
        int $mecToolB = `checkBox -q -v mecGuiBifgToolB`;
        int $mecMin = `checkBox -q -v mecGuiBifgMin`;
        int $mecMax = `checkBox -q -v mecGuiBifgMax`;
        int $mecMenuB = `checkBox -q -v mecGuiBifgMenuB`;

	string $mecStrFinal =	"if(`window -exists " + $mecObjName + "`)\n\t\t" +
		"deleteUI " + $mecObjName + ";\n" +
	"if( `windowPref -exists " + $mecObjName +"` )\n\t\t" +
		"windowPref -r " + $mecObjName + ";\n\n" +

	"window -w " + $mecWidth + " -h " + $mecHeight + " -t \"" + $mecTitle + "\" " 
        + "\n\t\t-titleBar " + $mecTitleB + " -mnb " + $mecMin + " -mxb " + $mecMax
        + " -mb " + $mecMenuB + " \n\t\t-tlb " + $mecToolB + " -sizeable " + $mecSizeable + " " + $mecObjName +";\n\n";

	$mecStrFinal += `scrollField -q -tx mecGuiBSf1`;
	

	$mecStrFinal += "\nshowWindow " + $mecObjName + ";\n";
	columnLayout -w 620 -h 550;
	scrollField -wordWrap true -w 600 -h 500
		-tx $mecStrFinal mecGuiBSfViewCode;

	showWindow mecWinBViewCodeWin;
}

global proc mecGuiBReload()
{
       if( `text -q -l mecGuiBError` != "None")
       {
           scrollField -e -cl mecGuiBSf1;
           scrollField -e -ip 1 -it `text -q -l mecGuiBReload` mecGuiBSf1;
       }

}

// Window Pref
//   This allows you to actively change what the window will look like.  This option
//      area is different from the other because this menu is accessed every time a
//      window is regenerated.  Unlike the other menu which just generate some code.
global proc mecGuiBWindowPref( string $mecParent)
{
	
	
	columnLayout ;
	text -l "" -h 10;
	textFieldGrp -l "Title" -h 25 -text "Generated Window" -cw2 75 225 mecGuiBtfgTitle;

	rowLayout -numberOfColumns 2 -columnWidth2 150 150;

	intFieldGrp -l "Width" -h 25 -cw2 75 40 -v1 300 mecGuiBifgWidth;
	intFieldGrp -l "Height" -cw2 75 40 -v1 300 mecGuiBifgHeight;

	setParent..;
        rowLayout -w 150 -nc 2 -columnWidth2 75 75 ;
        separator -w 75 -h 25 -hr 1 -st "none";

        	checkBox -l "Absolute" -v 1 -w 75 
                         -ann "w/h value will reflect the values above else resizeing the window will change w/h above." mecGuiBifgRelative;
        
        setParent..;
        	textFieldGrp -l "ObjName" -cw2 75 225 -text "mecGuiBGeneratedWin" mecGuiBWinObj;
                
        	
                //intFieldGrp -l "Sizeable" -cw2 75 40 -v1 1 mecGuiBifgSizeable;
        text -l "" -w 300 -h 15;
        rowLayout -w 300 -nc 4 -columnWidth4 75 75 25 75 ;
        separator -w 25 -hr 1 -st "none";
        checkBox -w 75 -l "Sizeable" -v 1
        		-align "center" mecGuiBifgSizeable;
        separator -w 25 -hr 1 -st "none";
        
        checkBox -w 75 -l "Toolbox"
        		-align "center" mecGuiBifgToolB;

        
        setParent..;
        rowLayout -w 300 -nc 4 -columnWidth4 75 75 25 75 ;
        separator -w 25 -hr 1 -st "none";
        checkBox -w 75 -l "Minimize" -v 1
        		-align "center" mecGuiBifgMin;

        separator -w 25 -hr 1 -st "none";
        
        checkBox -w 75 -l "Maximize" -v 1
        		-align "center" mecGuiBifgMax;
        
        setParent..;
        rowLayout -w 300 -nc 4 -columnWidth4 75 75 25 75;
        separator -w 25 -hr 1 -st "none";
        checkBox -w 75 -l "Menu Bar"  -v 1
        		-align "center" mecGuiBifgMenuB;
        
        separator -w 25 -hr 1 -st "none";
        checkBox -w 75 -l "Title Bar" -v 1
        		-align "center" mecGuiBifgTitleB;
        setParent $mecParent;
}

// work for the column layout frame
global proc mecGuiBLayoutsColWork()
{
	// After hitting apply button on the ColumnLayout button.
	//   Info will be grabed stored into a string and then sent to the scrollField.
	string $mecFinalStr = "columnLayout ";
	
	string $mecObjName = `textField -q -text mecGuiBLayoutsColTf1`;
	int $mecWidth = `intField -q -v mecGuiBLayoutsColTf2`;
	int $mecHeight = `intField -q -v mecGuiBLayoutsColTf3`;
	string $mecParent = `textField -q -text mecGuiBLayoutsColTf4`;

	//checking to see if the checkbox is checked.

	if(`checkBox -q -v mecGuiBLayoutsColCk2 `)
	{
		$mecFinalStr += "-w " + $mecWidth + " ";
	}

	if(`checkBox -q -v mecGuiBLayoutsColCk3`)
	{
		$mecFinalStr += "-h " + $mecHeight + " ";
	}

	if(`checkBox -q -v mecGuiBLayoutsColCk4`)
	{
		$mecFinalStr += "-p " + $mecParent + " ";
	}

	if(`checkBox -q -v mecGuiBLayoutsColCk1`)
	{
		$mecFinalStr += $mecObjName;
	}
	
	$mecFinalStr += ";\n";
	// inserting text into the scrollList
	scrollField -e -ip 0 -insertText $mecFinalStr mecGuiBSf1;

}

// Generation of the columnWidth fields.
global proc mecguiBLayoutsRowDynamic(string $mecParent)
{
	// Must set the parent so that the layout is going in the right spot.	
	setParent mecGuiBLayoutsRowDyColTop;
	deleteUI mecGuiBLayoutsRowDyCol;
	columnLayout mecGuiBLayoutsRowDyCol;
	
	//Dynamicly Generated controls.
	//ColumnWidth flag
	$mecNC = `intField -q -v mecGuiBLayoutsRowTf5`;
	if( $mecNC > 0 )
	{	
		int $mecI = 1;
		for( $mecI; $mecI <= $mecNC; $mecI++  )
		{
			string $mecName = "columnWidth" + $mecI;
			rowLayout -nc 3 -columnWidth3 100 60 135;
			text -l $mecName;
			string $mecObj = "mecGuiBLayoutsRowCW" + $mecI;
			intField -w 40 $mecObj;
			text -l "" -w 135;
			setParent..;
		}
	}
	
	setParent $mecParent;

}

global proc mecGuiGridLayoutWork()
{
        int $mecWidth = `intField -q -v mecGuiBGridLayoutObj1`;
        int $mecHeight = `intField -q -v mecGuiBGridLayoutObj2`;
        string $mecParent = `textField -q -tx mecGuiBGridLayoutObj3`;
        string $mecObjName = `textFieldGrp -q -tx mecGuiBGridLayoutObj4`;
        int $mecNR = `intField -q -v mecGuiBGridLayoutObj5`;
        int $mecNC = `intField -q -v mecGuiBGridLayoutObj6`;
        int $mecCR = `checkBox -q -v mecGuiBGridLayoutCk6`;
        int $mecAEC = `checkBox -q -v mecGuiBGridLayoutCk7`;
        int $mecAG = `checkBox -q -v mecGuiBGridLayoutCk8`;
        int $mecCW = `intField -q -v mecGuiBGridLayoutObj7`;
        int $mecCH = `intField -q -v mecGuiBGridLayoutObj8`;
        string $mecFinalStr = "gridLayout ";

        //Checking width
        if(`checkBox -q -v mecGuiBGridLayoutCk1`)
             $mecFinalStr += "-w " + $mecWidth + " ";
        
        //Checking Height
        if(`checkBox -q -v mecGuiBGridLayoutCk2`)
             $mecFinalStr += "-h " + $mecHeight + " ";

        //Checking Parents
        if(`checkBox -q -v mecGuiBGridLayoutCk3`)
             $mecFinalStr += "-p \"" + $mecParent + "\" ";

        //Checking nr
        if(`checkBox -q -v mecGuiBGridLayoutCk4`)
             $mecFinalStr += "-nr " + $mecNR + " ";

        //Checking nc
        if(`checkBox -q -v mecGuiBGridLayoutCk5`)
             $mecFinalStr += "-nc " + $mecNC + " ";

        $mecFinalStr += "\n-cr " + $mecCR + " -aec " + $mecAEC + " -ag " + $mecAG + "\n";

        //Checking Cell Width
        if(`checkBox -q -v mecGuiBGridLayoutCk9`)
             $mecFinalStr += "-cw " + $mecCW + " ";

        //Checking Cell Height
        if(`checkBox -q -v mecGuiBGridLayoutCk10`)
             $mecFinalStr += "-ch " + $mecCH + " ";
             
        $mecFinalStr += " " + $mecObjName + ";\n";
        
        scrollField -e -ip 0 -insertText $mecFinalStr mecGuiBSf1;
        

}

global proc mecGuiBFormLayoutWork2()
{
       string $mecAttach = `radioCollection -q -sl mecGuiBFormLayoutRCollection`;
       string $mecControl = `textFieldGrp -q -tx mecGuiBFormLayoutObj8`;
       string $mecEdge = `optionMenu -q -v mecGuiBFormLayoutObj9`;

       int $mecPosition = `intField -q -v mecGuiBFormLayoutObj11`;
       string $mecAttControl = `textFieldGrp -q -tx mecGuiBFormLayoutObj12`;
       string $mecMainObj = `textFieldGrp -q -tx mecGuiBFormLayoutObj13`;
       
       string $mecFinalStr = "formLayout -e ";

       //if Attach form
       if($mecAttach == "mecGuiBFormLayoutRB1")
       {
           $mecFinalStr += "-af " + $mecControl + " \"" + $mecEdge + "\" " + $mecOffset + " ";
       }

       //if attach opposite form
       if($mecAttach == "mecGuiBFormLayoutRB2")
       {
           $mecFinalStr += "-aof " + $mecControl + " \"" + $mecEdge + "\" " + $mecOffset + " ";
       }

       //if attach control
       if($mecAttach == "mecGuiBFormLayoutRB3")
       {
           $mecFinalStr += "-ac " + $mecControl + " \"" + $mecEdge + "\" " + $mecOffset + " " + $mecAttControl + " ";
       }

       //if attach opposite form
       if($mecAttach == "mecGuiBFormLayoutRB4")
       {
           $mecFinalStr += "-aoc "  + $mecControl + " \"" + $mecEdge + "\" " + $mecOffset + " " + $mecAttControl + " ";
       }

       //if attach position
       if($mecAttach == "mecGuiBFormLayoutRB5")
       {
           $mecFinalStr += "-ap " + $mecControl + " \"" + $mecEdge + "\" " + $mecOffset + " " + $mecPosition + " ";
       }

       //if attach None
       if($mecAttach == "mecGuiBFormLayoutRB6")
       {
           $mecFinalStr += "-an " + $mecControl + " \"" + $mecEdge + "\" ";
       }

       $mecFinalStr += $mecMainObj + ";\n";
       
       scrollField -e -ip 0 -insertText $mecFinalStr mecGuiBSf1;

}

global proc mecGuiBGroupsFFieldGrpWork()

          string $mecParent = `textField -q -tx mecGuiBGroupsFFieldGrpObj3`;
          string $mecAnn = `textField -q -tx mecGuiBGroupsFFieldGrpObj21`;
          string $mecLabel = `textFieldGrp -q -tx mecGuiBGroupsFFieldGrpObj20`;
          string $mecExtra = `textField -q -tx mecGuiBGroupsFFieldGrpObj5`;
          int $mecCC = `checkBox -q -v mecGuiBGroupsFFieldGrpCk5`;
          int $mecDC = `checkBox -q -v mecGuiBGroupsFFieldGrpCk6`;
          string $mecObjName = `textFieldGrp -q -tx mecGuiBGroupsFFieldGrpObj6`;
          int $mecNumF = `intField -q -v mecGuiBGroupsFFieldGrpObj4`;


          string $mecFinalStr = "floatFieldGrp ";
          
          //Width
          if(`checkBox -q -v mecGuiBGroupsFFieldGrpCk1`)
                  $mecFinalStr += "-w " + $mecWidth + " ";

          //Height
          if(`checkBox -q -v mecGuiBGroupsFFieldGrpCk2`)
                  $mecFinalStr += "-h " + $mecHeight + " ";

          //Parent;
          if(`checkBox -q -v mecGuiBGroupsFFieldGrpCk3`)
                  $mecFinalStr += "-p " + $mecParent + " ";

          //Annotation
          if(`checkBox -q -v mecGuiBGroupsFFieldGrpCk10`)
                  $mecFinalStr += "\n\t\t-ann \"" + $mecAnn + "\"\n\t\t";

          //Label
                  $mecFinalStr += "-l \"" + $mecLabel + "\" ";

          //Extra label
          if(`checkBox -q -v mecGuiBGroupsFFieldGrpCk4`)
                  $mecFinalStr += "-el \"" + $mecExtra + "\" ";

          //number of Fields
          $mecFinalStr += "-nf " + $mecNumF + " ";

          
          //Value
          int $mecI = 1;
          for($mecI; $mecI <= $mecNumF; $mecI++)
          {
                     string $mecTemp =  "mecGuiBFFieldGrpDynV" + $mecI;
                     float $mecTemp2 = `floatField -q -v $mecTemp`;
                     $mecFinalStr += "-v" + $mecI + " " + $mecTemp2 + " ";
          }
          
          //ColumnWidth

          //Making exception for extraLabel
          int $mecNumF2 = $mecNumF + 1;
          if(`checkBox -q -v mecGuiBGroupsFFieldGrpCk4`)
              $mecNumF2++;
              

          $mecFinalStr += "\n\t\t-cw" + $mecNumF2 + " ";
          int $mecM = 1;
          
          for($mecM; $mecM <= $mecNumF2; $mecM++)
          {
                     string $mecTemp = "mecGuiBFFieldGrpDynCW" + $mecM;
                     int $mecTemp2 = `intField -q -v $mecTemp`;
                     $mecFinalStr += $mecTemp2 + " ";
          }


          //ChangeCommand
          if(`checkBox -q -v mecGuiBGroupsFFieldGrpCk5` )
                $mecFinalStr += "-cc \"\" \n\t\t";

          //DragCommand
          if(`checkBox -q -v mecGuiBGroupsFFieldGrpCk6`)
                $mecFinalStr += "-dc \"\" \n\t\t";

          //ObjectName
                $mecFinalStr += $mecObjName + ";\n";

           scrollField -e -ip 0 -insertText $mecFinalStr mecGuiBSf1;

 
}
